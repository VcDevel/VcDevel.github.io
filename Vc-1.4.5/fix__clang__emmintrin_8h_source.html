43#ifdef _mm_shuffle_epi32
44#undef _mm_shuffle_epi32
45#define _mm_shuffle_epi32(a, imm) __extension__ ({ \
46 (__m128i)__builtin_shufflevector((__v4si)(__m128i)(a), (__v4si) _mm_set1_epi32(0), \
47 (imm) & 0x3, ((imm) & 0xc) >> 2, \
48 ((imm) & 0x30) >> 4, ((imm) & 0xc0) >> 6); })
49#endif
50
51#ifdef _mm_shufflelo_epi16
52#undef _mm_shufflelo_epi16
53#define _mm_shufflelo_epi16(a, imm) __extension__ ({ \
54 (__m128i)__builtin_shufflevector((__v8hi)(__m128i)(a), (__v8hi) _mm_set1_epi16(0), \
55 (imm) & 0x3, ((imm) & 0xc) >> 2, \
56 ((imm) & 0x30) >> 4, ((imm) & 0xc0) >> 6, \
57 4, 5, 6, 7); })
58#endif
59
60#ifdef _mm_shufflehi_epi16
61#undef _mm_shufflehi_epi16
62#define _mm_shufflehi_epi16(a, imm) __extension__ ({ \
63 (__m128i)__builtin_shufflevector((__v8hi)(__m128i)(a), (__v8hi) _mm_set1_epi16(0), \
64 0, 1, 2, 3, \
65 4 + (((imm) & 0x03) >> 0), \
66 4 + (((imm) & 0x0c) >> 2), \
67 4 + (((imm) & 0x30) >> 4), \
68 4 + (((imm) & 0xc0) >> 6)); })
69#endif
70
71#ifdef _mm_shuffle_pd
72#undef _mm_shuffle_pd
73#define _mm_shuffle_pd(a, b, i) __extension__ ({ \
74 __builtin_shufflevector((__m128d)(a), (__m128d)(b), (i) & 1, (((i) & 2) >> 1) + 2); })
75#endif
